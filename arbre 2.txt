### Exemple de Code pour une Fonctionnalité de Base

Voici un exemple de code pour une fonctionnalité de base de gestion des comptes :

```python
class Compte:
    def __init__(self, numero, nom, type_compte, classe_compte):
        self.numero = numero
        self.nom = nom
        self.type_compte = type_compte
        self.classe_compte = classe_compte
        self.solde = 0.0
        self.ecritures = []

    def crediter(self, montant, ecriture):
        self.solde += montant
        self.ecritures.append(ecriture)

    def debiter(self, montant, ecriture):
        if self.solde >= montant:
            self.solde -= montant
            self.ecritures.append(ecriture)
        else:
            raise ValueError("Solde insuffisant")

    def afficher_solde(self):
        print(f"Le solde du compte {self.numero} est de {self.solde}")

    def afficher_ecritures(self):
        for ecriture in self.ecritures:
            print(f"Date: {ecriture.date}, Libellé: {ecriture.libelle}, Montant: {ecriture.montant}")

# Exemple d'utilisation
compte1 = Compte("12345", "Compte Courant", "Actif", "Classe A")
compte1.crediter(1000.0, {"date": "2023-10-01", "libelle": "Vente de produits", "montant": 1000.0})
compte1.debiter(500.0, {"date": "2023-10-02", "libelle": "Paiement vente", "montant": 500.0})
compte1.afficher_solde()
compte1.afficher_ecritures()
```

### Exemple de Test Unitaire

Voici un exemple de test unitaire pour la classe `Compte` :

```python
import unittest

class TestCompte(unittest.TestCase):
    def setUp(self):
        self.compte = Compte("12345", "Compte Courant", "Actif", "Classe A")

    def test_crediter(self):
        self.compte.crediter(1000.0, {"date": "2023-10-01", "libelle": "Vente de produits", "montant": 1000.0})
        self.assertEqual(self.compte.solde, 1000.0)

    def test_debiter(self):
        self.compte.crediter(1000.0, {"date": "2023-10-01", "libelle": "Vente de produits", "montant": 1000.0})
        self.compte.debiter(500.0, {"date": "2023-10-02", "libelle": "Paiement vente", "montant": 500.0})
        self.assertEqual(self.compte.solde, 500.0)

    def test_debiter_solde_insuffisant(self):
        with self.assertRaises(ValueError):
            self.compte.debiter(500.0, {"date": "2023-10-02", "libelle": "Paiement vente", "montant": 500.0})

if __name__ == '__main__':
    unittest.main()
